from qiskit import *
qc=QuantumCircuit(8,2)
qc.x(0)
qc.x(1)
qc.x(2)
qc.barrier()
qc.ccx(0,1,3)
qc.barrier()
qc.ccx(0,1,3)
qc.barrier()
qc.cx(2,5)
qc.cx(4,5)
qc.barrier()
qc.ccx(2,4,6)
qc.barrier()
qc.measure(5,0)
qc.measure(7,1)
qc.draw(output="mpl")
counts=execute(qc,Aer.get_backend("qasm_simulator")).result().get_counts()
print(counts)
